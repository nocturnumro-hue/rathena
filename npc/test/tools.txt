// --------------------------
// Load Inventory (Online/Offline)
// --------------------------
function	script	F_LoadInventory	{
	.@cid = getarg(0);
	.@online = getarg(1, 0);

	deletearray @inventory_id[0], getarraysize(@inventory_id);
	deletearray @inventory_amount[0], getarraysize(@inventory_amount);
	set @inventory_count, 0;

	if (.@cid == getcharid(3) || (.@online && attachrid(.@cid))) {
		getinventorylist;
		copyarray @inventory_id, @inventorylist_id, @inventorylist_count;
		copyarray @inventory_amount, @inventorylist_amount, @inventorylist_count;
		set @inventory_count, @inventorylist_count;
		if (.@online && .@cid != getcharid(3)) detachrid;
	} else {
		query_sql("SELECT `nameid`, `amount` FROM `inventory` WHERE `char_id` = "+.@cid+" AND `equip` = 0", @inventory_id, @inventory_amount);
		set @inventory_count, getarraysize(@inventory_id);
	}

	return @inventory_count;
}
